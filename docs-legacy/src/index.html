Installation & Usage 
# Install via npm
npm install @meridian-ux/components

# Or via yarn
yarn add @meridian-ux/components
// Import in your app
import '@meridian-ux/components';

// Or import specific components
import '@meridian-ux/components/mx-button';
import '@meridian-ux/components/mx-input';
<!-- Use in your HTML -->
<mx-button type="primary">Click Me</mx-button>
<mx-input placeholder="Enter text..." />
<mx-card title="Hello World">
  <p>Card content goes here</p>
</mx-card>
